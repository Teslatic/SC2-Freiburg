35 class ReplayBuffer(object):
36  ''' Experience Replay Buffer Class '''
37  def __init__(self, capacity):
38  self.capacity = capacity
39  self.memory = []
40  self.position = 0
41 
42  def __len__(self):
43  return len(self.memory)
44 
45  def __getitem__(self, key):
46  return self.memory[key]
47 
48  def push(self, *args):
49  ''' store transition in replay buffer '''
50  if len(self.memory) < self.capacity:
51  self.memory.append(None)
52  self.memory[self.position] = Transition(*args)
53  self.position = (self.position+1) % self.capacity
54 
55  def sample(self, batch_size):
56  ''' return random sample transition from ER '''
57  return random.sample(self.memory, batch_size)

230  def optimize(self):
231  if len(self.memory) < self._batch_size:
232  return
233  transitions = self.memory.sample(self._batch_size)
234 
235  batch = Transition(*zip(*transitions))
236  non_final_mask = torch.tensor(tuple(map(lambda s: s is not None,
237  batch.next_state)), device=self._device, dtype=torch.uint8)
238  non_final_next_states = torch.cat([s for s in batch.next_state if s is not None])
239 
240  state_batch = torch.cat(batch.state)
241  action_batch = torch.cat(batch.action).unsqueeze(1)
242  reward_batch = torch.cat(batch.reward)
243 
244  q_action = self._net(state_batch).gather(1, action_batch)
245 
246  q_action_next = torch.zeros(self._batch_size, device=self._device)
247  non_final_next_action_q = self._target_net(non_final_next_states)
248 
249  q_action_next[non_final_mask] = non_final_next_action_q.max(1)[0].detach()
250 
251  td_target_actions = (q_action_next * self._gamma) + reward_batch
